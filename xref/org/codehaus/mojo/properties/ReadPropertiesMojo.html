41  /**
42   * The read-project-properties goal reads property files and URLs and stores the properties as project properties. It
43   * serves as an alternate to specifying properties in pom.xml. It is especially useful when making properties defined in
44   * a runtime resource available at build time.
45   *
46   * @author <a href="mailto:zarars@gmail.com">Zarar Siddiqi</a>
47   * @author <a href="mailto:Krystian.Nowak@gmail.com">Krystian Nowak</a>
48   */
49  @Mojo( name = "read-project-properties", defaultPhase = LifecyclePhase.NONE, threadSafe = true )
50  public class ReadPropertiesMojo
51      extends AbstractMojo
52  {
53      @Parameter( defaultValue = "${project}", readonly = true, required = true )
54      private MavenProject project;
55  
56      /**
57       * The properties files that will be used when reading properties.
58       */
59      @Parameter
60      private File[] files = new File[0];
61  
62      /**
63       * @param files The files to set for tests.
64       */
65      public void setFiles( File[] files )
66      {
67          if ( files == null )
68          {
69              this.files = new File[0];
70          }
71          else
72          {
73              this.files = new File[files.length];
74              System.arraycopy( files, 0, this.files, 0, files.length );
75          }
76      }
77  
78      /**
79       * The URLs that will be used when reading properties. These may be non-standard URLs of the form
80       * <code>classpath:com/company/resource.properties</code>. Note that the type is not <code>URL</code> for this
81       * reason and therefore will be explicitly checked by this Mojo.
82       */
83      @Parameter
84      private String[] urls = new String[0];
85  
86      /**
87       * Default scope for test access.
88       * 
89       * @param urls The URLs to set for tests.
90       */
91      public void setUrls( String[] urls )
92      {
93          if ( urls == null )
94          {
95              this.urls = null;
96          }
97          else
98          {
99              this.urls = new String[urls.length];
100             System.arraycopy( urls, 0, this.urls, 0, urls.length );
101         }
102     }
103 
104     /**
105      * If the plugin should be quiet if any of the files was not found
106      */
107     @Parameter( defaultValue = "false" )
108     private boolean quiet;
109 
110     /**
111      * Prefix that will be added before name of each property.
112      * Can be useful for separating properties with same name from different files.
113      */
114     @Parameter
115     private String keyPrefix = null;
116 
117     public void setKeyPrefix( String keyPrefix )
118     {
119         this.keyPrefix = keyPrefix;
120     }
121 
122     /**
123      * Used for resolving property placeholders.
124      */
125     private final PropertyResolver resolver = new PropertyResolver();
126 
127     /** {@inheritDoc} */
128     public void execute()
129         throws MojoExecutionException, MojoFailureException
130     {
131         checkParameters();
132 
133         loadFiles();
134 
135         loadUrls();
136 
137         resolveProperties();
138     }
139 
140     private void checkParameters()
141         throws MojoExecutionException
142     {
143         if ( files.length > 0 && urls.length > 0 )
144         {
145             throw new MojoExecutionException( "Set files or URLs but not both - otherwise "
146                 + "no order of precedence can be guaranteed" );
147         }
148     }
149 
150     private void loadFiles()
151         throws MojoExecutionException
152     {
153         for ( File file : files )
154         {
155             load( new FileResource( file ) );
156         }
157     }
158 
159     private void loadUrls()
160         throws MojoExecutionException
161     {
162         for ( String url : urls )
163         {
164             load( new UrlResource( url ) );
165         }
166     }
167 
168     private void load( Resource resource )
169         throws MojoExecutionException
170     {
171         if ( resource.canBeOpened() )
172         {
173             loadProperties( resource );
174         }
175         else
176         {
177             missing( resource );
178         }
179     }
180 
181     private void loadProperties( Resource resource )
182         throws MojoExecutionException
183     {
184         try
185         {
186             getLog().debug( "Loading properties from " + resource );
187 
188             try ( InputStream stream = resource.getInputStream() )
189             {
190                 if ( keyPrefix != null )
191                 {
192                     Properties properties = new Properties();
193                     properties.load( stream );
194                     Properties projectProperties = project.getProperties();
195                     for ( String key : properties.stringPropertyNames() )
196                     {
197                         projectProperties.put( keyPrefix + key, properties.get( key ) );
198                     }
199                 }
200                 else
201                 {
202                     project.getProperties().load( stream );
203                 }
204             }
205         }
206         catch ( IOException e )
207         {
208             throw new MojoExecutionException( "Error reading properties from " + resource, e );
209         }
210     }
211 
212     private void missing( Resource resource )
213         throws MojoExecutionException
214     {
215         if ( quiet )
216         {
217             getLog().info( "Quiet processing - ignoring properties cannot be loaded from " + resource );
218         }
219         else
220         {
221             throw new MojoExecutionException( "Properties could not be loaded from " + resource );
222         }
223     }
224 
225     private void resolveProperties()
226         throws MojoExecutionException, MojoFailureException
227     {
228         Properties environment = loadSystemEnvironmentPropertiesWhenDefined();
229         Properties projectProperties = project.getProperties();
230 
231         for ( Enumeration<?> n = projectProperties.propertyNames(); n.hasMoreElements(); )
232         {
233             String k = (String) n.nextElement();
234             projectProperties.setProperty( k, getPropertyValue( k, projectProperties, environment ) );
235         }
236     }
237 
238     private Properties loadSystemEnvironmentPropertiesWhenDefined()
239         throws MojoExecutionException
240     {
241         Properties projectProperties = project.getProperties();
242 
243         boolean useEnvVariables = false;
244         for ( Enumeration<?> n = projectProperties.propertyNames(); n.hasMoreElements(); )
245         {
246             String k = (String) n.nextElement();
247             String p = (String) projectProperties.get( k );
248             if ( p.contains( "${env." ) )
249             {
250                 useEnvVariables = true;
251                 break;
252             }
253         }
254         Properties environment = null;
255         if ( useEnvVariables )
256         {
257             try
258             {
259                 environment = getSystemEnvVars();
260             }
261             catch ( IOException e )
262             {
263                 throw new MojoExecutionException( "Error getting system environment variables: ", e );
264             }
265         }
266         return environment;
267     }
268 
269     private String getPropertyValue( String k, Properties p, Properties environment )
270         throws MojoFailureException
271     {
272         try
273         {
274             return resolver.getPropertyValue( k, p, environment );
275         }
276         catch ( IllegalArgumentException e )
277         {
278             throw new MojoFailureException( e.getMessage() );
279         }
280     }

283      * Override-able for test purposes.
284      * 
285      * @return The shell environment variables, can be empty but never <code>null</code>.

314     private static abstract class Resource
315     {
316         private InputStream stream;
317 
318         public abstract boolean canBeOpened();
319 
320         protected abstract InputStream openStream()
321             throws IOException;
322 
323         public InputStream getInputStream()
324             throws IOException
325         {
326             if ( stream == null )
327             {
328                 stream = openStream();
329             }
330             return stream;
331         }
332     }
333 
334     private static class FileResource
335         extends Resource
336     {
337         private final File file;
338 
339         public FileResource( File file )
340         {
341             this.file = file;
342         }
343 
344         public boolean canBeOpened()
345         {
346             return file.exists();
347         }
348 
349         protected InputStream openStream()
350             throws IOException
351         {
352             return new BufferedInputStream( new FileInputStream( file ) );
353         }
354 
355         public String toString()
356         {
357             return "File: " + file;
358         }
359     }
360 
361     private static class UrlResource
362         extends Resource
363     {
364         private static final String CLASSPATH_PREFIX = "classpath:";
365 
366         private static final String SLASH_PREFIX = "/";
367 
368         private final URL url;
369 
370         private boolean isMissingClasspathResouce = false;
371 
372         private String classpathUrl;
373 
374         public UrlResource( String url )
375             throws MojoExecutionException
376         {
377             if ( url.startsWith( CLASSPATH_PREFIX ) )
378             {
379                 String resource = url.substring( CLASSPATH_PREFIX.length() );
380                 if ( resource.startsWith( SLASH_PREFIX ) )
381                 {
382                     resource = resource.substring( 1 );
383                 }
384                 this.url = getClass().getClassLoader().getResource( resource );
385                 if ( this.url == null )
386                 {
387                     isMissingClasspathResouce = true;
388                     classpathUrl = url;
389                 }
390             }
391             else
392             {
393                 try
394                 {
395                     this.url = new URL( url );
396                 }
397                 catch ( MalformedURLException e )
398                 {
399                     throw new MojoExecutionException( "Badly formed URL " + url + " - " + e.getMessage() );
400                 }
401             }
402         }
403 
404         public boolean canBeOpened()
405         {
406             if ( isMissingClasspathResouce )
407             {
408                 return false;
409             }
410             try
411             {
412                 openStream();
413             }
414             catch ( IOException e )
415             {
416                 return false;
417             }
418             return true;
419         }
420 
421         protected InputStream openStream()
422             throws IOException
423         {
424             return new BufferedInputStream( url.openStream() );
425         }
426 
427         public String toString()
428         {
429             if ( !isMissingClasspathResouce )
430             {
431                 return "URL " + url.toString();
432             }
433             return classpathUrl;
434         }
435     }
